\section*{Independent Random Distributions}

\textbf{Expectation:}
\[
E[aX + bY + c] = aE[X] + bE[Y] + c
\]

\textbf{Variance:}
\[
\mathrm{Var}(aX + bY + c) = a^2 \mathrm{Var}(X) + b^2 \mathrm{Var}(Y)
\]

\textbf{(Key: independence $\Rightarrow$ no covariance term)}

\textbf{General form (not independent):}
\[
\mathrm{Var}(aX + bY) = a^2 \mathrm{Var}(X) + b^2 \mathrm{Var}(Y) + 2ab\,\mathrm{Cov}(X,Y)
\]

\textbf{Expectation (always linear):}
\[
E[aX + bY + c] = aE[X] + bE[Y] + c
\]

\section*{Variance of Difference of Sample Means (Dependent Samples)}

Let $\bar{X}_1, \bar{X}_2$ be sample means from samples of size $n_1, n_2$.

\textbf{General (dependent case):}
\[
\mathrm{Var}(\bar{X}_1 - \bar{X}_2)
=
\frac{\sigma_1^2}{n_1}
+
\frac{\sigma_2^2}{n_2}
-
2\,\mathrm{Cov}(\bar{X}_1, \bar{X}_2)
\]

\textbf{Covariance of sample means:}
\[
\mathrm{Cov}(\bar{X}_1, \bar{X}_2)
=
\frac{1}{n_1 n_2}
\sum_{i=1}^{n_1} \sum_{j=1}^{n_2}
\mathrm{Cov}(X_i, Y_j)
\]

\textbf{Special case (paired samples, $n_1 = n_2 = n$):}
\[
\mathrm{Var}(\bar{X}_1 - \bar{X}_2)
=
\frac{\sigma_1^2}{n}
+
\frac{\sigma_2^2}{n}
-
2\frac{\mathrm{Cov}(X_1, X_2)}{n}
\]

\textbf{If independent:}
\[
\mathrm{Cov}(\bar{X}_1, \bar{X}_2)=0
\Rightarrow
\mathrm{Var}(\bar{X}_1 - \bar{X}_2)
=
\frac{\sigma_1^2}{n_1}
+
\frac{\sigma_2^2}{n_2}
\]

\section*{Joint Distributions}

\textbf{Joint PMF (discrete):}
\[
f_{X,Y}(x,y) = P(X=x, Y=y)
\]

\textbf{Properties:}
\[
f_{X,Y}(x,y) \ge 0, \quad \sum_x \sum_y f_{X,Y}(x,y)=1
\]

\textbf{Joint PDF (continuous):}
\[
P(a<X\le b, c<Y\le d) = \int_a^b \int_c^d f_{X,Y}(x,y)\,dy\,dx
\]

\textbf{Marginals:}
\[
f_X(x)=\sum_y f_{X,Y}(x,y) \quad \text{or} \quad f_X(x)=\int f_{X,Y}(x,y)\,dy
\]
\[
f_Y(y)=\sum_x f_{X,Y}(x,y) \quad \text{or} \quad f_Y(y)=\int f_{X,Y}(x,y)\,dx
\]

\section*{Conditional Distributions}

\[
f_{Y|X}(y|x) = \frac{f_{X,Y}(x,y)}{f_X(x)}
\]
\[
f_{X|Y}(x|y) = \frac{f_{X,Y}(x,y)}{f_Y(y)}
\]

\section*{Conditional Expectation}

\[
E[Y|X=x] = \sum_y y f_{Y|X}(y|x) \quad \text{(discrete)}
\]
\[
E[Y|X=x] = \int y f_{Y|X}(y|x)\,dy \quad \text{(continuous)}
\]

\textbf{Law of Iterated Expectation:}
\[
E[Y] = E[E[Y|X]]
\]

\section*{Independence}

\[
X \perp Y \iff f_{X,Y}(x,y)=f_X(x)f_Y(y)
\]
\[
P(X\le x, Y\le y)=P(X\le x)P(Y\le y)
\]

\section*{Covariance \& Correlation}

\[
\mathrm{Cov}(X,Y)=E[(X-\mu_X)(Y-\mu_Y)]
\]
\[
\mathrm{Cov}(X,Y)=E[XY]-E[X]E[Y]
\]

\[
\rho_{X,Y} = \frac{\mathrm{Cov}(X,Y)}{\sigma_X \sigma_Y}
\]

\textbf{Independence $\Rightarrow$ Cov = 0 (not converse)}

\section*{Sampling Distributions}

\textbf{Definition:} Distribution of a statistic

\textbf{Sample Mean:}
\[
E[\bar{X}] = \mu, \quad SD(\bar{X}) = \frac{\sigma}{\sqrt{n}}
\]

\textbf{If population normal:}
\[
\bar{X} \sim N\left(\mu, \frac{\sigma^2}{n}\right)
\]

\textbf{Sample Proportion:}
\[
E[\hat{p}] = p, \quad SD(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

\section*{Central Limit Theorem (CLT)}

\textbf{For mean (large $n \ge 30$):}
\[
\bar{X} \approx N\left(\mu, \frac{\sigma^2}{n}\right)
\]
\[
Z = \frac{\bar{X} - \mu}{\sigma/\sqrt{n}}
\]

\textbf{For proportion:}
\[
\hat{p} \approx N\left(p, \frac{p(1-p)}{n}\right)
\]
(valid if $np \ge 15$, $n(1-p) \ge 15$)

\section*{Standard Error (SE)}

\[
SE(\bar{X}) = \frac{\sigma}{\sqrt{n}} \quad \text{or} \quad \frac{S}{\sqrt{n}}
\]
\[
SE(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

\section*{Confidence Intervals (General)}

\[
CI = \text{Estimator} \pm (\text{Critical Value})(SE)
\]

\textbf{Margin of Error:}
\[
MoE = (\text{Critical Value})(SE)
\]

\section*{CI for Mean ($\sigma$ known)}

\[
\bar{X} \pm z_{\alpha/2} \frac{\sigma}{\sqrt{n}}
\]

\section*{CI for Mean ($\sigma$ unknown)}

\[
\bar{X} \pm t_{\alpha/2, n-1} \frac{S}{\sqrt{n}}
\]

\section*{CI for Proportion}

\[
\hat{p} \pm z_{\alpha/2} \sqrt{\frac{\hat{p}(1-\hat{p})}{n}}
\]

\section*{Confidence Interval Definition}

A $100(1-\alpha)\%$ CI satisfies:
\[
P(L(X_1,...,X_n) \le \theta \le U(X_1,...,X_n)) = 1-\alpha
\]

\textbf{Interpretation:}
\newline
In repeated sampling, $(1-\alpha)$ proportion of intervals contain $\theta$.

\section*{Key Notes}

\begin{itemize}
\item CI length = $2 \times MoE$
\item Larger $n$ $\Rightarrow$ smaller CI
\item Higher confidence $\Rightarrow$ wider CI
\item CLT requires large $n$ or normal population
\end{itemize}
