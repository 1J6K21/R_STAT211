\section*{Joint Distributions}

\textbf{Joint PMF (discrete):}
\[
p_{X,Y}(x,y)=P(X=x,Y=y)
\]
\[
\sum_x \sum_y p_{X,Y}(x,y)=1
\]

\textbf{Joint PDF (continuous):}
\[
f_{X,Y}(x,y)
\]
\[
\int\int f_{X,Y}(x,y)\,dx\,dy=1
\]

\textbf{Marginal Distributions}

\textbf{How to find marginals from joint:}

Discrete:
\[
p_X(x)=\sum_y p_{X,Y}(x,y)
\]
\[
p_Y(y)=\sum_x p_{X,Y}(x,y)
\]

Continuous:
\[
f_X(x)=\int_{-\infty}^{\infty} f_{X,Y}(x,y)\,dy
\]
\[
f_Y(y)=\int_{-\infty}^{\infty} f_{X,Y}(x,y)\,dx
\]

---

\section*{Conditional Distributions}

\textbf{Definition (discrete):}
\[
p_{Y|X}(y|x)=\frac{p_{X,Y}(x,y)}{p_X(x)}
\]

\textbf{Definition (continuous):}
\[
f_{Y|X}(y|x)=\frac{f_{X,Y}(x,y)}{f_X(x)}
\]

\textbf{How to find conditional density step-by-step:}

1. Start with joint density $f_{X,Y}(x,y)$  
2. Find marginal $f_X(x)$ by integrating out $y$  
3. Divide:
\[
f_{Y|X}(y|x)=\frac{f_{X,Y}(x,y)}{f_X(x)}
\]

Similarly,
\[
f_{X|Y}(x|y)=\frac{f_{X,Y}(x,y)}{f_Y(y)}
\]

---

\section*{Independence}

$X$ and $Y$ are independent if:

Discrete:
\[
p_{X,Y}(x,y)=p_X(x)p_Y(y)
\]

Continuous:
\[
f_{X,Y}(x,y)=f_X(x)f_Y(y)
\]

If independent:
\[
E[XY]=E[X]E[Y]
\]

---

\section*{Expectation}

\textbf{Discrete:}
\[
E[X]=\sum_x x p_X(x)
\]

\textbf{Continuous:}
\[
E[X]=\int_{-\infty}^{\infty} x f_X(x)\,dx
\]

\textbf{Function of RV:}
\[
E[g(X)]=\sum g(x)p_X(x)
\]
\[
E[g(X)]=\int g(x)f_X(x)\,dx
\]

\textbf{Linearity of Expectation}
\[
E[aX+bY+c]=aE[X]+bE[Y]+c
\]

---

\section*{Variance}

\textbf{Definition:}
\[
\text{Var}(X)=E[(X-\mu)^2]
\]
Simplified:
\[
\text{Var}(X)=E[X^2] - \mu_x^2
\]

\textbf{Shortcut Formula:}
\[
\text{Var}(X)=E[X^2]-(E[X])^2
\]

\[
\text{Var}(aX+b)=a^2\text{Var}(X)
\]

\textbf{If independent:}
\[
\text{Var}(aX+bY+c)=a^2\text{Var}(X)+b^2\text{Var}(Y)
\]

---

\section*{Covariance}

\textbf{Definition:}
\[
\text{Cov}(X,Y)=E[(X-\mu_X)(Y-\mu_Y)]
\]

\textbf{Computational Formula:}
\[
\text{Cov}(X,Y)=E[XY]-E[X]E[Y]
\]
Simplified:
\[
\text{Cov}(X,Y)=E[XY]-\mu _x \mu _y
\]

\textbf{General form (not independent):}
\[
\text{Var}(aX+bY)=a^2\text{Var}(X)+b^2\text{Var}(Y)+2ab\,\text{Cov}(X,Y)
\]

If independent:
\[
\text{Cov}(X,Y)=0
\]

---

\section*{Correlation}

\[
\rho_{X,Y}=\frac{\text{Cov}(X,Y)}{\sigma_X\sigma_Y}
\]

\[
-1 \le \rho \le 1
\]

---

\section*{Conditional Expectation}

\[
E[Y|X=x]=\sum_y y\,p_{Y|X}(y|x)
\]

\[
E[Y|X=x]=\int y\,f_{Y|X}(y|x)\,dy
\]

\textbf{Law of Total Expectation}
\[
E[Y]=E[E[Y|X]]
\]

---

\section*{Law of Total Variance}

\[
\text{Var}(Y)=E[\text{Var}(Y|X)]+\text{Var}(E[Y|X])
\]

---

\section*{Transformations of Random Variables}

If $Y=g(X)$ and $g$ is monotone:

\[
f_Y(y)=f_X(g^{-1}(y))\left|\frac{d}{dy}g^{-1}(y)\right|
\]

\textbf{Expectation and Variance under transformation}

\[
E[aX+b]=aE[X]+b
\]
\[
\text{Var}(aX+b)=a^2\text{Var}(X)
\]

---

\section*{CDF and PDF Relationship}

\[
F_X(x)=P(X\le x)
\]

If continuous:
\[
f_X(x)=\frac{d}{dx}F_X(x)
\]

\[
F_X(x)=\int_{-\infty}^{x} f_X(t)\,dt
\]

---

\section*{Sampling Distributions}

Let $X_1,\dots,X_n$ i.i.d.

\textbf{Sample Mean:}
\[
\bar X=\frac{1}{n}\sum_{i=1}^n X_i
\]

\[
E[\bar X]=\mu, \quad \text{Var}(\bar X)=\frac{\sigma^2}{n}, \quad SD(\bar X)=\frac{\sigma}{\sqrt n}
\]

\textbf{If population normal:}
\[
\bar{X} \sim N\left(\mu, \frac{\sigma^2}{n}\right)
\]

\textbf{Sample Proportion:}
\[
E[\hat{p}] = p, \quad SD(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

---

\section*{Variance of Diff. of Sample Means (Dependent)}

Let $\bar{X}_1, \bar{X}_2$ be sample means from samples of size $n_1, n_2$.

\textbf{General (dependent case):}
\[
\text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2} - 2\,\text{Cov}(\bar{X}_1, \bar{X}_2)
\]

\textbf{Covariance of sample means:}
\[
\text{Cov}(\bar{X}_1, \bar{X}_2) = \frac{1}{n_1 n_2} \sum_{i=1}^{n_1} \sum_{j=1}^{n_2} \text{Cov}(X_i, Y_j)
\]

\textbf{Special case (paired samples, $n_1 = n_2 = n$):}
\[
\text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n} + \frac{\sigma_2^2}{n} - 2\frac{\text{Cov}(X_1, X_2)}{n}
\]

\textbf{Sample SD of differences ($s_D$) shortcut:}
\[
s_D = \sqrt{s_X^2 + s_Y^2 - (2 \cdot r \cdot s_X \cdot s_Y)}
\]

\textbf{If independent:}
\[
\text{Cov}(\bar{X}_1, \bar{X}_2)=0 \implies \text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2}
\]

---

\section*{Central Limit Theorem}

\textbf{For mean (large $n \ge 30$):}
\[
Z = \frac{\bar X-\mu}{\sigma/\sqrt n}\approx N(0,1)
\]

\[
\bar X \approx N\left(\mu,\frac{\sigma^2}{n}\right)
\]

\textbf{For proportion:}
\[
\hat{p} \approx N\left(p, \frac{p(1-p)}{n}\right)
\]
(valid if $np \ge 15$, $n(1-p) \ge 15$)

---

\section*{Empirical Estimators}

Sample mean:
\[
\bar x=\frac{1}{n}\sum x_i
\]

Sample variance:
\[
s^2=\frac{1}{n-1}\sum (x_i-\bar x)^2
\]

Sample covariance:
\[
s_{XY}=\frac{1}{n-1}\sum (x_i-\bar x)(y_i-\bar y)
\]

Sample correlation:
\[
r=\frac{s_{XY}}{s_X s_Y}
\]

---

\section*{Bayes' Theorem (General Form)}

\[
P(A|B)=\frac{P(B|A)P(A)}{P(B)}
\]

Continuous:
\[
f_{X|Y}(x|y)=\frac{f_{Y|X}(y|x)f_X(x)}{f_Y(y)}
\]

---

---

\section*{Standard Error (SE)}

\[
SE(\bar{X}) = \frac{\sigma}{\sqrt{n}} \quad \text{or} \quad \frac{S}{\sqrt{n}}
\]
\[
SE(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

---

\section*{Confidence Intervals (General)}

\[
CI = \text{Estimator} \pm (\text{Critical Value})(SE)
\]

\textbf{Margin of Error:}
\[
MoE = (\text{Critical Value})(SE)
\]

A $100(1-\alpha)\%$ CI satisfies:
\[
P(L(X_1,\dots,X_n) \le \theta \le U(X_1,\dots,X_n)) = 1-\alpha
\]

\textbf{Interpretation:}
In repeated sampling, $(1-\alpha)$ proportion of intervals contain $\theta$.

---

\section*{Specific CIs}

\textbf{CI for Mean ($\sigma$ known):}
\[
\bar{X} \pm z_{\alpha/2} \frac{\sigma}{\sqrt{n}}
\]

\textbf{CI for Mean ($\sigma$ unknown):}
\[
\bar{X} \pm t_{\alpha/2, n-1} \frac{S}{\sqrt{n}}
\]

\textbf{CI for Proportion:}
\[
\hat{p} \pm z_{\alpha/2} \sqrt{\frac{\hat{p}(1-\hat{p})}{n}}
\]

\textbf{Key Notes:}
\begin{itemize}
\item CI length = $2 \times MoE$
\item Larger $n \Rightarrow$ smaller CI
\item Higher confidence $\Rightarrow$ wider CI
\item CLT requires large $n$ or normal population
\end{itemize}

---

\section*{Sample Size for Estimating $p$}

Solving for length $L$ of $(1-\alpha)100\%$ CI:
\[
n = \frac{4z_{\alpha/2}^2 \hat{p}(1-\hat{p})}{L^2}
\]
(\textit{Round up to next integer. $\hat{p}$ often unknown before sample.})

\textbf{If prior $p=p_0$ is known:}
\[
n = \frac{4z_{\alpha/2}^2 p_0(1-p_0)}{L^2}
\]

\textbf{If no prior info on $p$ is available:}
\[
n = \frac{z_{\alpha/2}^2}{L^2}
\]

\section*{Independence \& Conditional Properties}

\textbf{Additional Independence Facts:}

If $X, Y$ are independent ($X \perp Y$):
\begin{itemize}
    \item $f_{Y|X}(y|x) = f_Y(y)$ \quad (conditional = marginal)
    \item $E[Y|X] = E[Y]$
    \item $Var(Y|X) = Var(Y)$
\end{itemize}

$\Rightarrow$ Conditional distribution does \textbf{NOT} depend on $X$.

\textbf{Functions of independent RVs:} \\
If $X \perp Y$, then $g(X) \perp h(Y)$.

\textbf{Uncorrelated vs Independent:}
\begin{itemize}
    \item Independence $\Rightarrow Cov(X,Y)=0$
    \item $Cov(X,Y)=0$ does \textbf{NOT} $\Rightarrow$ independence
    \item \textit{Counterexample:} $X \sim N(0,1), Y = X^2$
\end{itemize}

\hrule

\textbf{Equivalent Conditions for Independence:}
\begin{itemize}
    \item Joint PMF/PDF: $f_{X,Y}(x,y) = f_X(x)f_Y(y)$
    \item Joint CDF: $F_{X,Y}(x,y) = F_X(x)F_Y(y)$
    \item Probabilities: $P(X \in A, Y \in B) = P(X \in A)P(Y \in B)$
\end{itemize}

\hrule

\textbf{Conditional Expectation (General Function):}
\[ E[g(X)|Y=y] = \sum_x g(x)p_{X|Y}(x|y) \quad \]\\ 
\[ \text{or} \quad \int g(x)f_{X|Y}(x|y) \, dx \]

$E[g(X)|Y]$ is a function of $Y \Rightarrow$ random variable.

\section*{Covariance \& Correlation (Advanced)}

\textbf{Covariance Properties:}
\begin{itemize}
    \item $Cov(X, c) = 0$
    \item $Cov(aX + b, cY + d) = ac \, Cov(X,Y)$
\end{itemize}

\textbf{Correlation:}
\[ \rho = \frac{Cov(X,Y)}{\sigma_X \sigma_Y} \]

\textbf{Properties:}
\begin{itemize}
    \item Unit-free
    \item $-1 \leq \rho \leq 1$
\end{itemize}

\textbf{Effect of Scaling:} \\
$Corr(aX, bY)$ is:
\begin{itemize}
    \item Same sign as $\rho$ if $ab > 0$
    \item Flips sign of $\rho$ if $ab < 0$
\end{itemize}

\textbf{Perfect Correlation:}
\begin{itemize}
    \item $\rho = 1 \iff Y = aX + b, a > 0$
    \item $\rho = -1 \iff Y = aX + b, a < 0$
\end{itemize}

\textbf{Sample Correlation:}
\begin{itemize}
    \item Unit-free
    \item $-1 \leq r \leq 1$
\end{itemize}

\hrule

\section*{Normal Distribution Moments}

If $Z \sim N(0,1)$:

\textbf{Odd moments:} $E[Z^{2k+1}] = 0$

\textbf{First four moments:}
\begin{itemize}
    \item $E[Z] = 0$
    \item $E[Z^2] = 1$
    \item $E[Z^3] = 0$
    \item $E[Z^4] = 3$
\end{itemize}

\section*{Conditional Variance}

\textbf{Definition:}
\[ Var(Y|X=x) = E[(Y - E[Y|X=x])^2 | X=x] \]

\textbf{Shortcut:}
\[ Var(Y|X) = E[Y^2 | X] - (E[Y|X])^2 \]

\textbf{Law of Iterated Expectation (Product):}
\[ E[XY] = E[ X E[Y|X] ] \]

\hrule

% Sampling & Estimator Properties
% =========================
\section*{Sampling \& Estimator Properties}

\textbf{Bias:}
\[ Bias(\hat{\theta}) = E[\hat{\theta}] - \theta \]

\textbf{Unbiased:}
\begin{itemize}
    \item $E[\bar{X}] = \mu$
    \item $E[\hat{p}] = p$
\end{itemize}

\textbf{Standard Error:}
\begin{itemize}
    \item $SE(\bar{X}) = \sigma/\sqrt{n}$ ($\sigma$ known)
    \item $SE(\bar{X}) = s/\sqrt{n}$ ($\sigma$ unknown)
    \item $SE(\hat{p}) = \sqrt{p(1-p)/n}$ (testing)
    \item $SE(\hat{p}) = \sqrt{\hat{p}(1-\hat{p})/n}$ (CI)
\end{itemize}

\textbf{CLT for proportions:} $np \geq 15, n(1-p) \geq 15$

\hrule

\section*{Confidence Intervals (Deeper)}

\textbf{Interpretation:}
\begin{itemize}
    \item \textbf{Before sampling:} $P(\theta \in CI) = 1 - \alpha$
    \item \textbf{After sampling:} $\theta$ is fixed $\Rightarrow$ probability is 0 or 1
    \item \textbf{Long-run:} $(1-\alpha)\%$ of intervals contain $\theta$
\end{itemize}

\textbf{Trade-off:}
\begin{itemize}
    \item Higher confidence $\Rightarrow$ wider interval
    \item Larger $n \Rightarrow$ narrower interval
\end{itemize}

\textbf{Sample Size for Mean ($\sigma$ known):}
\[ n = \left( \frac{z_{\alpha/2} \sigma}{MoE} \right)^2 \quad \text{(Round UP)} \]

\hrule

\section*{Hypothesis Testing Framework}

\textbf{Hypotheses:} $H_0$ (null), $H_a$ (alternative)

\textbf{Test Statistic:}
\[ \frac{\text{statistic} - \text{null value}}{SE} \]

\textbf{$p$-value:} $P(\text{result as extreme} \mid H_0 \text{ true})$

\textbf{Decision Rule:}
\begin{itemize}
    \item $p \leq \alpha \Rightarrow$ Reject $H_0$
    \item $p > \alpha \Rightarrow$ Fail to reject $H_0$
\end{itemize}

\hrule

% =========================
% Specific Tests
% =========================
\section*{Specific Tests}

\textbf{Z-test (mean, $\sigma$ known):} $Z = \frac{\bar{X} - \mu_0}{\sigma/\sqrt{n}}$

\textbf{T-test ($\sigma$ unknown):} $T = \frac{\bar{X} - \mu_0}{s/\sqrt{n}}, \, df = n-1$

\textbf{Z-test (proportion):} $Z = \frac{\hat{p} - p_0}{\sqrt{p_0(1-p_0)/n}}$

\hrule

% =========================
% Errors \& Power
% =========================
\section*{Errors \& Power}

\textbf{Type I Error:} Reject $H_0$ when true. \\
\textbf{Type II Error:} Fail to reject $H_0$ when false.

\textbf{Power:} $Power = 1 - \beta = P(\text{reject } H_0 \mid H_0 \text{ false})$

\hrule

\section*{Paired Data}

Let $D = X_1 - X_2$:
\begin{itemize}
    \item $E[D] = \mu_D$
    \item $Var(D) = \sigma_1^2 + \sigma_2^2 - 2Cov(X_1,X_2)$
\end{itemize}

\textbf{Paired t-test:} $T = \frac{\bar{D} - \mu_{D,0}}{s_D / \sqrt{n}}, \, df = n - 1$

\hrule

\section*{Ultra-Condensed Logic \& Edge Cases}

\textbf{Independence Checklist:}
\begin{itemize}
    \item If $f_{Y|X}(y|x)$ depends on $x \Rightarrow$ \textbf{NOT} independent
    \item If $E[Y|X]$ depends on $X \Rightarrow$ \textbf{NOT} independent
    \item If $Var(Y|X)$ depends on $X \Rightarrow$ \textbf{NOT} independent
\end{itemize}
\textbf{Fast test:} Check if $f_{X,Y}(x,y) = f_X(x)f_Y(y)$.

\hrule

\textbf{Covariance \& Expectation Tricks:}
\begin{itemize}
    \item $E[aX + bY] = aE[X] + bE[Y]$
    \item $Cov(X,Y) = 0 \centernot\implies$ independence (unless jointly normal)
    \item If $(X,Y)$ jointly normal: $Cov(X,Y)=0 \Rightarrow$ independence
\end{itemize}

\hrule

\textbf{Conditional Expectation Tricks:}
\textbf{Tower Property (extended):}
\[ E[E[g(X,Y)|X]] = E[g(X,Y)] \]
\textit{Special case:} $E[E[Y|X]] = E[Y]$

\hrule

\textbf{Variance Decomposition Insight:}
\textbf{Law of Total Variance:}
\[ Var(Y) = E[Var(Y|X)] + Var(E[Y|X]) \]
\textit{Interpretation:} Total variance = "within variability" + "between variability"

\hrule

\section*{Hypothesis Testing Shortcuts}

\textbf{Test Statistic Pattern:}
\[ \text{Test Stat} = \frac{\text{Estimator} - \text{Null}}{SE} = \frac{\text{signal}}{\text{noise}} \]

\textbf{Rejection Regions ($Z/T$):}
\begin{itemize}
    \item \textbf{Right-tailed:} reject if $stat > critical$
    \item \textbf{Left-tailed:} reject if $stat < -critical$
    \item \textbf{Two-sided:} reject if $|stat| > critical$
\end{itemize}

\textbf{$p$-value shortcuts:}
\begin{itemize}
    \item \textbf{Right tail:} $P(Z > z)$
    \item \textbf{Left tail:} $P(Z < z)$
    \item \textbf{Two-sided:} $2P(Z > |z|)$
\end{itemize}

\hrule

\section*{Confidence Interval Shortcuts}

\textbf{CI $\leftrightarrow$ Test Link:}
Reject $H_0: \mu = \mu_0 \iff \mu_0 \notin CI$

\textbf{Margin of Error ($MoE$):}
\[ MoE = (\text{critical value}) \times (SE) \]
\[ \text{CI width} = 2 \times MoE \]

\textbf{Quick intuition:}
\begin{itemize}
    \item Bigger $n \Rightarrow \downarrow SE \Rightarrow$ tighter CI
    \item Bigger $\sigma \Rightarrow \uparrow SE \Rightarrow$ wider CI
\end{itemize}

\hrule

% =========================
% Proportion Nuances

\section*{Proportion Nuances}

\textbf{Testing vs CI:}
\begin{itemize}
    \item \textbf{Test} uses $p_0$ in $SE = \sqrt{p_0(1-p_0)/n}$
    \item \textbf{CI} uses $\hat{p}$ in $SE = \sqrt{\hat{p}(1-\hat{p})/n}$
\end{itemize}
\textit{Reason:} The test assumes $H_0$ is true.

\textbf{Worst-case variance:}
$p(1-p)$ is maximized at $p = 0.5$.
$\Rightarrow$ Conservative $n$ uses $0.25$ in the numerator.

\hrule

\section*{Paired Data Insight}

\textbf{Variance of the Difference ($D = X_1 - X_2$):}
\[ Var(D) = \sigma_1^2 + \sigma_2^2 - 2\rho\sigma_1\sigma_2 \]
\begin{itemize}
    \item $\rho > 0 \Rightarrow$ reduces variance $\Rightarrow$ tighter CI
    \item $\rho < 0 \Rightarrow$ increases variance
\end{itemize}

\section*{Final Exam Strategy}

\textbf{Common Mistakes to Avoid:}
\begin{itemize}
    \item Using $\hat{p}$ instead of $p_0$ in hypothesis tests.
    \item Forgetting $df = n-1$ in t-tests.
    \item Forgetting the factor of 2 in two-sided $p$-values.
    \item Forgetting to \textbf{round UP} for sample size $n$.
\end{itemize}
