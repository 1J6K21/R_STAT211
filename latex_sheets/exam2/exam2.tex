\section*{Joint Distributions}

\textbf{Joint PMF (discrete):}
\[
p_{X,Y}(x,y)=P(X=x,Y=y)
\]
\[
\sum_x \sum_y p_{X,Y}(x,y)=1
\]

\textbf{Joint PDF (continuous):}
\[
f_{X,Y}(x,y)
\]
\[
\int\int f_{X,Y}(x,y)\,dx\,dy=1
\]

\textbf{Marginal Distributions}

\textbf{How to find marginals from joint:}

Discrete:
\[
p_X(x)=\sum_y p_{X,Y}(x,y)
\]
\[
p_Y(y)=\sum_x p_{X,Y}(x,y)
\]

Continuous:
\[
f_X(x)=\int_{-\infty}^{\infty} f_{X,Y}(x,y)\,dy
\]
\[
f_Y(y)=\int_{-\infty}^{\infty} f_{X,Y}(x,y)\,dx
\]

---

\section*{Conditional Distributions}

\textbf{Definition (discrete):}
\[
p_{Y|X}(y|x)=\frac{p_{X,Y}(x,y)}{p_X(x)}
\]

\textbf{Definition (continuous):}
\[
f_{Y|X}(y|x)=\frac{f_{X,Y}(x,y)}{f_X(x)}
\]

\textbf{How to find conditional density step-by-step:}

1. Start with joint density $f_{X,Y}(x,y)$  
2. Find marginal $f_X(x)$ by integrating out $y$  
3. Divide:
\[
f_{Y|X}(y|x)=\frac{f_{X,Y}(x,y)}{f_X(x)}
\]

Similarly,
\[
f_{X|Y}(x|y)=\frac{f_{X,Y}(x,y)}{f_Y(y)}
\]

---

\section*{Independence}

$X$ and $Y$ are independent if:

Discrete:
\[
p_{X,Y}(x,y)=p_X(x)p_Y(y)
\]

Continuous:
\[
f_{X,Y}(x,y)=f_X(x)f_Y(y)
\]

If independent:
\[
E[XY]=E[X]E[Y]
\]

---

\section*{Expectation}

\textbf{Discrete:}
\[
E[X]=\sum_x x p_X(x)
\]

\textbf{Continuous:}
\[
E[X]=\int_{-\infty}^{\infty} x f_X(x)\,dx
\]

\textbf{Function of RV:}
\[
E[g(X)]=\sum g(x)p_X(x)
\]
\[
E[g(X)]=\int g(x)f_X(x)\,dx
\]

\textbf{Linearity of Expectation}
\[
E[aX+bY+c]=aE[X]+bE[Y]+c
\]

---

\section*{Variance}

\textbf{Definition:}
\[
\text{Var}(X)=E[(X-\mu)^2]
\]
Simplified:
\[
\text{Var}(X)=E[X^2] - \mu_x^2
\]

\textbf{Shortcut Formula:}
\[
\text{Var}(X)=E[X^2]-(E[X])^2
\]

\[
\text{Var}(aX+b)=a^2\text{Var}(X)
\]

\textbf{If independent:}
\[
\text{Var}(aX+bY+c)=a^2\text{Var}(X)+b^2\text{Var}(Y)
\]

---

\section*{Covariance}

\textbf{Definition:}
\[
\text{Cov}(X,Y)=E[(X-\mu_X)(Y-\mu_Y)]
\]

\textbf{Computational Formula:}
\[
\text{Cov}(X,Y)=E[XY]-E[X]E[Y]
\]
Simplified:
\[
\text{Cov}(X,Y)=E[XY]-\mu _x \mu _y
\]

\textbf{General form (not independent):}
\[
\text{Var}(aX+bY)=a^2\text{Var}(X)+b^2\text{Var}(Y)+2ab\,\text{Cov}(X,Y)
\]

If independent:
\[
\text{Cov}(X,Y)=0
\]

---

\section*{Correlation}

\[
\rho_{X,Y}=\frac{\text{Cov}(X,Y)}{\sigma_X\sigma_Y}
\]

\[
-1 \le \rho \le 1
\]

---

\section*{Conditional Expectation}

\[
E[Y|X=x]=\sum_y y\,p_{Y|X}(y|x)
\]

\[
E[Y|X=x]=\int y\,f_{Y|X}(y|x)\,dy
\]

\textbf{Law of Total Expectation}
\[
E[Y]=E[E[Y|X]]
\]

---

\section*{Law of Total Variance}

\[
\text{Var}(Y)=E[\text{Var}(Y|X)]+\text{Var}(E[Y|X])
\]

---

\section*{Transformations of Random Variables}

If $Y=g(X)$ and $g$ is monotone:

\[
f_Y(y)=f_X(g^{-1}(y))\left|\frac{d}{dy}g^{-1}(y)\right|
\]

\textbf{Expectation and Variance under transformation}

\[
E[aX+b]=aE[X]+b
\]
\[
\text{Var}(aX+b)=a^2\text{Var}(X)
\]

---

\section*{CDF and PDF Relationship}

\[
F_X(x)=P(X\le x)
\]

If continuous:
\[
f_X(x)=\frac{d}{dx}F_X(x)
\]

\[
F_X(x)=\int_{-\infty}^{x} f_X(t)\,dt
\]

\section*{Sampling Distributions}

Let $X_1,\dots,X_n$ i.i.d.

\textbf{Sample Mean:}
\[
\bar X=\frac{1}{n}\sum_{i=1}^n X_i
\]

\[
E[\bar X]=\mu, \quad \text{Var}(\bar X)=\frac{\sigma^2}{n}, \quad SD(\bar X)=\frac{\sigma}{\sqrt n}
\]

\textbf{If population normal:}
\[
\bar{X} \sim N\left(\mu, \frac{\sigma^2}{n}\right)
\]

\textbf{Sample Proportion:}
\[
E[\hat{p}] = p, \quad SD(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

---

\section*{Variance of Diff. of Sample Means (Dependent)}

Let $\bar{X}_1, \bar{X}_2$ be sample means from samples of size $n_1, n_2$.

\textbf{General (dependent case):}
\[
\text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2} - 2\,\text{Cov}(\bar{X}_1, \bar{X}_2)
\]

\textbf{Covariance of sample means:}
\[
\text{Cov}(\bar{X}_1, \bar{X}_2) = \frac{1}{n_1 n_2} \sum_{i=1}^{n_1} \sum_{j=1}^{n_2} \text{Cov}(X_i, Y_j)
\]

\textbf{Special case (paired samples, $n_1 = n_2 = n$):}
\[
\text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n} + \frac{\sigma_2^2}{n} - 2\frac{\text{Cov}(X_1, X_2)}{n}
\]

\textbf{Sample SD of differences ($s_D$) shortcut:}
\[
s_D = \sqrt{s_X^2 + s_Y^2 - (2 \cdot r \cdot s_X \cdot s_Y)}
\]

\textbf{If independent:}
\[
\text{Cov}(\bar{X}_1, \bar{X}_2)=0 \implies \text{Var}(\bar{X}_1 - \bar{X}_2) = \frac{\sigma_1^2}{n_1} + \frac{\sigma_2^2}{n_2}
\]

---

\section*{Central Limit Theorem}

\textbf{For mean (large $n \ge 30$):}
\[
Z = \frac{\bar X-\mu}{\sigma/\sqrt n}\approx N(0,1)
\]

\[
\bar X \approx N\left(\mu,\frac{\sigma^2}{n}\right)
\]

\textbf{For proportion:}
\[
\hat{p} \approx N\left(p, \frac{p(1-p)}{n}\right)
\]
(valid if $np \ge 15$, $n(1-p) \ge 15$)

---

\section*{Empirical Estimators}

Sample mean:
\[
\bar x=\frac{1}{n}\sum x_i
\]

Sample variance:
\[
s^2=\frac{1}{n-1}\sum (x_i-\bar x)^2
\]

Sample covariance:
\[
s_{XY}=\frac{1}{n-1}\sum (x_i-\bar x)(y_i-\bar y)
\]

Sample correlation:
\[
r=\frac{s_{XY}}{s_X s_Y}
\]

---

\section*{Bayes' Theorem (General Form)}

\[
P(A|B)=\frac{P(B|A)P(A)}{P(B)}
\]

Continuous:
\[
f_{X|Y}(x|y)=\frac{f_{Y|X}(y|x)f_X(x)}{f_Y(y)}
\]

\section*{Standard Error (SE)}

\[
SE(\bar{X}) = \frac{\sigma}{\sqrt{n}} \quad \text{or} \quad \frac{S}{\sqrt{n}}
\]
\[
SE(\hat{p}) = \sqrt{\frac{p(1-p)}{n}}
\]

---

\section*{Confidence Intervals (General)}

\[
CI = \text{Estimator} \pm (\text{Critical Value})(SE)
\]

\textbf{Margin of Error:}
\[
MoE = (\text{Critical Value})(SE)
\]

A $100(1-\alpha)\%$ CI satisfies:
\[
P(L(X_1,\dots,X_n) \le \theta \le U(X_1,\dots,X_n)) = 1-\alpha
\]

\textbf{Interpretation:}
In repeated sampling, $(1-\alpha)$ proportion of intervals contain $\theta$.

---

\section*{Specific CIs}

\textbf{CI for Mean ($\sigma$ known):}
\[
\bar{X} \pm z_{\alpha/2} \frac{\sigma}{\sqrt{n}}
\]

\textbf{CI for Mean ($\sigma$ unknown):}
\[
\bar{X} \pm t_{\alpha/2, n-1} \frac{S}{\sqrt{n}}
\]

\textbf{CI for Proportion:}
\[
\hat{p} \pm z_{\alpha/2} \sqrt{\frac{\hat{p}(1-\hat{p})}{n}}
\]

\textbf{Key Notes:}
\begin{itemize}
\item CI length = $2 \times MoE$
\item Larger $n \Rightarrow$ smaller CI
\item Higher confidence $\Rightarrow$ wider CI
\item CLT requires large $n$ or normal population
\end{itemize}

---

\section*{Sample Size for Estimating $p$}

Solving for length $L$ of $(1-\alpha)100\%$ CI:
\[
n = \frac{4z_{\alpha/2}^2 \hat{p}(1-\hat{p})}{L^2}
\]
(\textit{Round up to next integer. $\hat{p}$ often unknown before sample.})

\textbf{If prior $p=p_0$ is known:}
\[
n = \frac{4z_{\alpha/2}^2 p_0(1-p_0)}{L^2}
\]

\textbf{If no prior info on $p$ is available:}
\[
n = \frac{z_{\alpha/2}^2}{L^2}
\]
